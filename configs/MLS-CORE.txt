# MLS-CORE Configuration

Device role: Multilayer Core Switch
Model: Cisco 3560
Function: VLAN gateways, inter-VLAN routing, trunks, ACLs, DHCP relay and SSH management.

---

## Basic Configuration

```bash
hostname MLS-CORE

ip domain-name school.local

username admin privilege 15 secret <SECURE_PASSWORD>
enable secret <ENABLE_SECRET>

ip ssh version 2
```

---

## VLAN Configuration

```bash
vlan 10
 name ALUMNOS
exit

vlan 20
 name PROFESORES
exit

vlan 30
 name ADMINS
exit

vlan 40
 name INVITADOS
exit

vlan 50
 name SERVERS
exit

vlan 99
 name MGMT
exit
```

## Inter-VLAN Routing

```bash
ip routing
```

## SVI Interfaces

```bash
interface vlan 10
 description Gateway VLAN ALUMNOS
 ip address 192.168.0.1 255.255.254.0
 ip helper-address 192.168.2.98
 ip access-group ACL_ALUMNOS in
 no shutdown
exit

interface vlan 20
 description Gateway VLAN PROFESORES
 ip address 192.168.2.1 255.255.255.192
 ip helper-address 192.168.2.98

exit

interface vlan 40
```

## Link to Router

```bash

 no switchport

 no shutdown
exit
```

## Default Route

```bash
ip route 0.0.0.0 0.0.0.0 10.0.0.1
```

---

## ACL for Students

```bash
ip access-list extended ACL_ALUMNOS
 deny ip 192.168.0.0 0.0.1.255 192.168.2.96 0.0.0.7
 permit ip any any
exit
```

---

## ACL for Guests

```bash
ip access-list extended ACL_INVITADOS
 deny ip 192.168.2.64 0.0.0.31 192.168.0.0 0.0.3.255
 permit ip any any
exit
```

---

## SSH Access Restriction

```bash
ip access-list standard SSH_ADMINS_ONLY
 permit 192.168.2.112 0.0.0.7
 deny any
exit
```

---

## VTY Lines

```bash
line vty 0 15
 login local
 transport input ssh
 access-class SSH_ADMINS_ONLY in
exit
```
